function onOpen() {
  var ui = SpreadsheetApp.getUi();    
  // Menu pour la détection d'erreurs
  ui.createMenu("Détection Erreurs")
    .addItem("Détecter les erreurs de choix d'articles", "verifierErreurs")
    .addItem("Mise à jour des U9-U12 sans CDP", "envoyerRelances")
    .addToUi();
  
  // Menu pour la rétroaction
  ui.createMenu("Rétroaction")
    .addItem("Mise à jour données spordles", "convertirEtImporterExcel")    
    .addItem("Générer fichiers Rétroaction", "genererRetroComplet")  
    .addToUi();
  
  // Menu pour les rapports
  ui.createMenu("Rapports")
    .addItem("Générer les rapports", "genererRapportsComplets")
    .addToUi();

    // Menu paramètres
    ui.createMenu('Paramètres')
        .addItem('Ouvrir la configuration', 'ouvrirPopupParametres')
    .addToUi();

}
/**
 * Ouvre la popup de configuration des paramètres.
 */
function ouvrirPopupParametres() {
  var html = HtmlService.createHtmlOutputFromFile('Parametres')
      .setWidth(1024)
      .setHeight(720);
  SpreadsheetApp.getUi().showModalDialog(html, 'Configuration des paramètres');
}
